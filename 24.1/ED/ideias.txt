# (incompleto)

def bin_subarr_search(n, arr):
    l = len(arr) // 2

    if n < arr[l]:
        return bin_subarr_search(n, arr[:l])
    elif n > arr[l]:
        return bin_subarr_search(n, arr[l:])
    else:
        if n < arr[l // 2]:
            return bin_subarr_search(n, arr[l])
